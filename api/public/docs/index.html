    const url = new URL(
    "http://127.0.0.1:8000/api/register"
);

let headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

    "email": "email@gmail.com",
    "password": "my25strong*&password",
    "name": "John Doe"
}

fetch(url, {
    method: "POST",
    headers: headers,
    body: body
})
